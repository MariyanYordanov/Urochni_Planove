**Генериране на Adversarial Examples:**

**Fast Gradient Sign Method (FGSM) — метод на бързия знаков градиент:**
```python
import tensorflow as tf
import numpy as np

def generate_adversarial_example(model, image, true_label, epsilon=0.01):
    """
    Generate adversarial example using FGSM attack
    """
    image = tf.cast(image, tf.float32)
    
    with tf.GradientTape() as tape:
        tape.watch(image)
        prediction = model(image)
        loss = tf.keras.losses.categorical_crossentropy(true_label, prediction)
    
    # Calculate gradient
    gradient = tape.gradient(loss, image)
    
    # Generate adversarial example
    signed_grad = tf.sign(gradient)
    adversarial_image = image + epsilon * signed_grad
    adversarial_image = tf.clip_by_value(adversarial_image, 0, 1)
    
    return adversarial_image

# Example usage for image classification
def test_adversarial_robustness(model, test_images, test_labels):
    """
    Test model robustness against adversarial attacks
    """
    original_accuracy = 0
    adversarial_accuracy = 0
    
    for image, label in zip(test_images, test_labels):
        # Test original image
        original_pred = model.predict(np.expand_dims(image, axis=0))
        original_correct = np.argmax(original_pred) == np.argmax(label)
        
        # Generate и test adversarial example
        adv_image = generate_adversarial_example(model, image, label)
        adv_pred = model.predict(np.expand_dims(adv_image, axis=0))
        adv_correct = np.argmax(adv_pred) == np.argmax(label)
        
        original_accuracy += original_correct
        adversarial_accuracy += adv_correct
    
    total_samples = len(test_images)
    print(f"Original Accuracy: {original_accuracy/total_samples:.2%}")
    print(f"Adversarial Accuracy: {adversarial_accuracy/total_samples:.2%}")
    print(f"Attack Success Rate: {(original_accuracy-adversarial_accuracy)/total_samples:.2%}")
```

```python
def pgd_attack(model, image, true_label, epsilon=0.3, alpha=0.01, num_iter=40):
    """
    Projected Gradient Descent adversarial attack
    """
    image = tf.cast(image, tf.float32)
    original_image = tf.identity(image)
    
    for i in range(num_iter):
        with tf.GradientTape() as tape:
            tape.watch(image)
            prediction = model(image)
            loss = tf.keras.losses.categorical_crossentropy(true_label, prediction)
        
        gradient = tape.gradient(loss, image)
        
        # Take step в gradient direction
        image = image + alpha * tf.sign(gradient)
        
        # Project back to epsilon ball
        perturbation = image - original_image
        perturbation = tf.clip_by_value(perturbation, -epsilon, epsilon)
        image = original_image + perturbation
        image = tf.clip_by_value(image, 0, 1)
    
    return image

def c_w_attack(model, image, target_class, confidence=0, max_iterations=1000):
    """
    Carlini & Wagner (C&W) attack implementation
    """
    import scipy.optimize as so
    
    def objective_function(delta):
        adversarial_image = tf.clip_by_value(image + delta.reshape(image.shape), 0, 1)
        logits = model(tf.expand_dims(adversarial_image, axis=0))
        
        # C&W loss function
        target_logit = logits[0][target_class]
        max_other_logit = tf.reduce_max(logits[0][:target_class] + logits[0][target_class+1:])
        
        loss1 = tf.maximum(0.0, max_other_logit - target_logit + confidence)
        loss2 = tf.reduce_sum(tf.square(delta))
        
        return loss1 + 0.01 * loss2
    
    # Optimize perturbation
    initial_delta = np.zeros(image.shape).flatten()
    result = so.minimize(objective_function, initial_delta, method='L-BFGS-B')
    
    optimal_delta = result.x.reshape(image.shape)
    adversarial_image = tf.clip_by_value(image + optimal_delta, 0, 1)
    
    return adversarial_image
```

```python
def adversarial_training(model, train_data, epochs=10, epsilon=0.1):
    """
    Adversarial training для robust model development
    """
    optimizer = tf.keras.optimizers.Adam(learning_rate=0.001)
    
    for epoch in range(epochs):
        print(f"Epoch {epoch + 1}/{epochs}")
        
        for batch_x, batch_y in train_data:
            # Generate adversarial examples
            adversarial_batch = tf.map_fn(
                lambda x: generate_adversarial_example(model, x[0], x[1], epsilon),
                (batch_x, batch_y),
                dtype=tf.float32
            )
            
            # Combine clean и adversarial examples
            combined_x = tf.concat([batch_x, adversarial_batch], axis=0)
            combined_y = tf.concat([batch_y, batch_y], axis=0)
            
            # Train на combined dataset
            with tf.GradientTape() as tape:
                predictions = model(combined_x, training=True)
                loss = tf.keras.losses.categorical_crossentropy(combined_y, predictions)
                loss = tf.reduce_mean(loss)
            
            gradients = tape.gradient(loss, model.trainable_variables)
            optimizer.apply_gradients(zip(gradients, model.trainable_variables))

def defensive_distillation(teacher_model, student_model, train_data, temperature=20):
    """
    Defensive distillation за adversarial robustness
    """
    # Phase 1: Train teacher model на hard targets
    teacher_model.compile(optimizer='adam', loss='categorical_crossentropy')
    teacher_model.fit(train_data, epochs=10)
    
    # Phase 2: Generate soft targets using temperature scaling
    def temperature_softmax(logits, temperature):
        return tf.nn.softmax(logits / temperature)
    
    soft_targets = []
    for batch_x, _ in train_data:
        teacher_logits = teacher_model(batch_x)
        soft_target = temperature_softmax(teacher_logits, temperature)
        soft_targets.append(soft_target)
    
    # Phase 3: Train student model на soft targets
    student_model.compile(optimizer='adam', loss='categorical_crossentropy')
    
    for batch_x, soft_y in zip(train_data, soft_targets):
        student_model.train_on_batch(batch_x, soft_y)
    
    return student_model
```

```python
def adversarial_detection_system(model, input_data):
    """
    Multi-layered adversarial example detection system
    """
    
    def statistical_test_detection(image):
        """Statistical property analysis"""
        # Benford's law test for pixel value distribution
        pixel_values = image.flatten()
        first_digits = [int(str(abs(x))[0]) for x in pixel_values if x != 0]
        
        expected_benford = [30.1, 17.6, 12.5, 9.7, 7.9, 6.7, 5.8, 5.1, 4.6]
        observed_freq = [first_digits.count(i+1)/len(first_digits)*100 for i in range(9)]
        
        chi_square = sum((obs - exp)** 2 / exp for obs, exp in zip(observed_freq, expected_benford))
        return chi_square > 16.92  # Critical value for 8 degrees of freedom
    
    def neural_network_detection(image):
        """Dedicated detection neural network"""
        # Train separate network to classify clean vs adversarial
        detection_model = tf.keras.Sequential([
            tf.keras.layers.Conv2D(32, 3, activation='relu'),
            tf.keras.layers.MaxPooling2D(),
            tf.keras.layers.Conv2D(64, 3, activation='relu'),
            tf.keras.layers.GlobalAveragePooling2D(),
            tf.keras.layers.Dense(1, activation='sigmoid')
        ])
        
        prediction = detection_model.predict(np.expand_dims(image, axis=0))
        return prediction[0][0] > 0.5
    
    def prediction_inconsistency_detection(image):
        """Multiple model agreement analysis"""
        models = [model]  # Add ensemble of models
        predictions = [m.predict(np.expand_dims(image, axis=0)) for m in models]
        
        # Check prediction consistency
        predicted_classes = [np.argmax(pred) for pred in predictions]
        agreement_rate = len(set(predicted_classes)) / len(predicted_classes)
        
        return agreement_rate < 0.8  # Threshold for inconsistency
    
    # Combine detection methods
    is_statistical_anomaly = statistical_test_detection(input_data)
    is_neural_detected = neural_network_detection(input_data)
    is_inconsistent = prediction_inconsistency_detection(input_data)
    
    detection_score = sum([is_statistical_anomaly, is_neural_detected, is_inconsistent])
    
    return {
        'is_adversarial': detection_score >= 2,
        'confidence': detection_score / 3,
        'individual_detections': {
            'statistical': is_statistical_anomaly,
            'neural': is_neural_detected,
            'inconsistency': is_inconsistent
        }
    }
```

```python
import numpy as np
from sklearn.metrics import accuracy_score, precision_score, recall_score

def calculate_fairness_metrics(y_true, y_pred, sensitive_attribute):
    """
    Calculate various fairness metrics за AI model evaluation
    """
    # Get unique groups в sensitive attribute
    groups = np.unique(sensitive_attribute)
    
    fairness_metrics = {}
    
    for group in groups:
        group_mask = sensitive_attribute == group
        group_y_true = y_true[group_mask]
        group_y_pred = y_pred[group_mask]
        
        fairness_metrics[f'group_{group}'] = {
            'accuracy': accuracy_score(group_y_true, group_y_pred),
            'precision': precision_score(group_y_true, group_y_pred, average='weighted'),
            'recall': recall_score(group_y_true, group_y_pred, average='weighted'),
            'positive_rate': np.mean(group_y_pred),
            'size': len(group_y_true)
        }
    
    # Calculate fairness measures
    accuracies = [fairness_metrics[f'group_{g}']['accuracy'] for g in groups]
    positive_rates = [fairness_metrics[f'group_{g}']['positive_rate'] for g in groups]
    
    fairness_measures = {
        'demographic_parity': max(positive_rates) - min(positive_rates),
        'equalized_odds': max(accuracies) - min(accuracies),
        'statistical_significance': len(groups)
    }
    
    return fairness_metrics, fairness_measures

def bias_mitigation_preprocessing(X, y, sensitive_attribute):
    """
    Preprocessing bias mitigation using resampling
    """
    from sklearn.utils import resample
    
    # Identify majority и minority groups
    groups = np.unique(sensitive_attribute)
    group_sizes = [np.sum(sensitive_attribute == g) for g in groups]
    max_size = max(group_sizes)
    
    # Upsample minority groups
    X_resampled = []
    y_resampled = []
    sensitive_resampled = []
    
    for group in groups:
        group_mask = sensitive_attribute == group
        X_group = X[group_mask]
        y_group = y[group_mask]
        sensitive_group = sensitive_attribute[group_mask]
        
        # Resample to match majority group size
        X_group_resampled, y_group_resampled, sensitive_group_resampled = resample(
            X_group, y_group, sensitive_group,
            n_samples=max_size,
            random_state=42
        )
        
        X_resampled.append(X_group_resampled)
        y_resampled.append(y_group_resampled)
        sensitive_resampled.append(sensitive_group_resampled)
    
    return (np.vstack(X_resampled), 
            np.hstack(y_resampled), 
            np.hstack(sensitive_resampled))
```

```python
            device_keys = self._generate_device_key_pair(device_id)

            )

                'device_identity': {
                    'device_id': device_id,
                    'hardware_identity': hardware_identity,
                    'public_key': device_keys['public_key'],

            'authentication': self._authenticate_update_request(device_id, update_data),
            'authorization': self._authorize_update_operation(device_id, update_data),
            'integrity_verification': self._verify_update_integrity(update_data),
            'compatibility_check': self._check_update_compatibility(device_id, update_data),
            'rollback_preparation': self._prepare_update_rollback(device_id),
            'update_application': self._apply_secure_update(device_id, update_data),
            'verification': self._verify_update_success(device_id, update_data)

            'rollback_available': update_process['rollback_preparation']['success']
        }
    
    def _apply_secure_update(self, device_id, update_data):
        """Apply update с security validation"""
        try:
            # Pre-update state capture
            pre_update_state = self._capture_device_state(device_id)
            
            # Secure update delivery
            update_delivery = self._deliver_update_securely(device_id, update_data)
            
            if not update_delivery['success']:

            # Update application
            application_result = self._execute_update_application(device_id, update_data)
            
            # Post-update validation

                'validation': post_update_validation
            }
        
        except Exception as e:
            return {
```
