        """Calculate GDPR penalty risk"""
        annual_revenue = org_profile.get("annual_revenue", 0)
        records_affected = scenario.get("records_affected", 0)
        
        # GDPR penalty calculation
        revenue_penalty = annual_revenue * 0.04  # 4% of annual revenue
        fixed_penalty = 20000000  # €20 million
        
        maximum_penalty = max(revenue_penalty, fixed_penalty)
        
        # Risk factors that influence penalty severity
        aggravating_factors = scenario.get("aggravating_factors", [])
        mitigating_factors = scenario.get("mitigating_factors", [])
        
        # Calculate penalty probability and severity
        base_probability = 0.3  # 30% base probability for GDPR enforcement
        severity_multiplier = 0.1  # Start with 10% of maximum penalty
        
        # Adjust based on factors
        for factor in aggravating_factors:
            if factor == "willful_violation":
                severity_multiplier += 0.3
                base_probability += 0.2
            elif factor == "repeat_violation":
                severity_multiplier += 0.2
                base_probability += 0.3
            elif factor == "large_scale_breach":
                severity_multiplier += 0.1
                base_probability += 0.1
        
        for factor in mitigating_factors:
            if factor == "cooperation_with_authorities":
                severity_multiplier -= 0.05
            elif factor == "prompt_notification":
                severity_multiplier -= 0.03
            elif factor == "remediation_efforts":
                severity_multiplier -= 0.05
        
        # Ensure boundaries
        severity_multiplier = max(0.01, min(1.0, severity_multiplier))
        base_probability = max(0.1, min(0.9, base_probability))
        
        expected_penalty = maximum_penalty * severity_multiplier
        penalty_risk = expected_penalty * base_probability
        
        return {
            "regulation": "GDPR",
            "maximum_penalty": maximum_penalty,
            "expected_penalty": expected_penalty,
            "penalty_probability": base_probability,

            def withdraw_consent(self, data_subject_id, purpose):

                consent_id = self._generate_consent_id(data_subject_id, purpose)
                
                if consent_id in self.consent_records:
                    # Record withdrawal
                    withdrawal_record = {
                        "data_subject_id": data_subject_id,
                        "purpose": purpose,
                        "withdrawal_timestamp": datetime.datetime.now().isoformat(),
                        "original_consent": self.consent_records[consent_id]
                    }
                    
                    # Update consent record
                    self.consent_records[consent_id]["consent_withdrawn"] = True
                    self.consent_records[consent_id]["withdrawal_timestamp"] = withdrawal_record["withdrawal_timestamp"]

            def _validate_gdpr_consent(self, consent_record):
                """Validate consent against GDPR requirements"""
                validation_errors = []
                
                # Check if consent is freely given
                if not consent_record.get("freely_given", False):
                    validation_errors.append("Consent must be freely given")
                
                # Check if consent is specific
                if not consent_record.get("specific", False):
                    validation_errors.append("Consent must be specific to purpose")
                
                # Check if consent is informed
                if not consent_record.get("informed", False):
                    validation_errors.append("Consent must be informed")
                
                # Check if consent is unambiguous
                if not consent_record.get("unambiguous", False):
                    validation_errors.append("Consent must be unambiguous")
                
                # Check consent text clarity
                if not consent_record.get("consent_text") or len(consent_record["consent_text"]) < 50:

                    "valid": len(validation_errors) == 0,
