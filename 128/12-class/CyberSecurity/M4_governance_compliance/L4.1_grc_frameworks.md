pythonimport numpy as np
import pandas as pd
from scipy import stats

class QuantitativeRiskAnalysis:
    """
    Comprehensive quantitative cybersecurity risk analysis
    Uses statistical methods and financial modeling
    """
    
    def __init__(self):
        self.risk_register = {}
        self.asset_values = {}
        self.threat_probabilities = {}
        
    def calculate_annual_loss_expectancy(self, asset_value, threat_probability, impact_percentage):
        """
        Calculate Annual Loss Expectancy (ALE)
        ALE = Asset Value × Threat Probability × Impact Percentage
        """
        single_loss_expectancy = asset_value * impact_percentage
        ale = single_loss_expectancy * threat_probability
        
        return {
            "asset_value": asset_value,
            "single_loss_expectancy": single_loss_expectancy,
            "annual_loss_expectancy": ale,
            "threat_probability": threat_probability
        }
    
    def monte_carlo_risk_simulation(self, scenarios, iterations=10000):
        """
        Monte Carlo simulation for complex risk scenarios
        Accounts for uncertainty in probability and impact estimates
        """
        simulation_results = []
        
        for i in range(iterations):
            total_loss = 0
            for scenario in scenarios:
                # Sample from probability distributions
                probability = np.random.beta(
                    scenario['prob_alpha'], 
                    scenario['prob_beta']
                )
                impact = np.random.lognormal(
                    scenario['impact_mean'],
                    scenario['impact_std']
                )
                
                # Determine if event occurs
                if np.random.random() < probability:
                    total_loss += impact
                    
            simulation_results.append(total_loss)
        
        # Calculate risk metrics
        risk_metrics = {
            "mean_annual_loss": np.mean(simulation_results),
            "median_annual_loss": np.median(simulation_results),
            "var_95": np.percentile(simulation_results, 95),  # Value at Risk
            "cvar_95": np.mean([x for x in simulation_results if x >= np.percentile(simulation_results, 95)]),  # Conditional VaR
            "max_loss": np.max(simulation_results),
            "probability_of_loss": len([x for x in simulation_results if x > 0]) / iterations
        }
        
        return risk_metrics, simulation_results
    
    def cyber_risk_scenarios(self):

class QualitativeRiskAnalysis:
    """
    Qualitative risk analysis using expert judgment
    Useful when quantitative data is limited
    """
    
    def __init__(self):
        self.risk_matrix = self._create_risk_matrix()
        
    def _create_risk_matrix(self):
        """Create 5x5 risk matrix for qualitative assessment"""
        probability_levels = {
            1: "Very Low (0-5%)",
            2: "Low (6-25%)", 
            3: "Medium (26-50%)",
            4: "High (51-75%)",
            5: "Very High (76-100%)"
        }
        
        impact_levels = {
            1: "Minimal (<$50K)",
            2: "Minor ($50K-$500K)",
            3: "Moderate ($500K-$5M)", 
            4: "Major ($5M-$50M)",
            5: "Catastrophic (>$50M)"
        }
        
        # Risk level calculation: Probability × Impact
        risk_matrix = {}
        for prob in range(1, 6):
            for impact in range(1, 6):
                risk_score = prob * impact
                if risk_score <= 6:
                    risk_level = "Low"
                elif risk_score <= 12:
                    risk_level = "Medium"
                elif risk_score <= 16:
                    risk_level = "High"
                else:
                    risk_level = "Critical"
                    
                risk_matrix[(prob, impact)] = {
                    "risk_score": risk_score,
                    "risk_level": risk_level
                }
        
        return risk_matrix
    
    def assess_threat_scenario(self, threat_name, probability_level, impact_level):
        """Assess individual threat scenario"""
        risk_assessment = self.risk_matrix[(probability_level, impact_level)]
        
        return {
            "threat": threat_name,
            "probability": probability_level,
            "impact": impact_level,
            "risk_score": risk_assessment["risk_score"],
            "risk_level": risk_assessment["risk_level"]
        }

                    "Third party can better manage the risk"
                ],
                "requirements": [
                    "Due diligence on transfer mechanism",
                    "Contract terms review",
                    "Residual risk assessment"
                ]
            }
        }
    
    def optimize_risk_treatment_portfolio(self, identified_risks):
        """
        Optimize risk treatment decisions across portfolio
        Consider budget constraints and risk interdependencies
        """
        treatment_portfolio = {}
        
        for risk_id, risk_data in identified_risks.items():
            treatment_options = self._evaluate_treatment_options(risk_data)
            optimal_treatment = self._select_optimal_treatment(
                risk_data, 
                treatment_options
            )
            
            treatment_portfolio[risk_id] = {
                "risk_data": risk_data,
                "recommended_treatment": optimal_treatment,
                "alternatives": treatment_options,
                "rationale": optimal_treatment["selection_rationale"]
            }
        
        return treatment_portfolio
    
    def _evaluate_treatment_options(self, risk_data):
        """Evaluate all treatment options for specific risk"""
        options = {}
        
        # Accept option
        options["accept"] = {
            "residual_risk": risk_data["inherent_risk"],
            "cost": 0,
            "feasibility": "High",
            "effectiveness": 0
        }
        
        # Mitigation options
        for control in risk_data.get("potential_controls", []):
            options[f"mitigate_{control['name']}"] = {
                "residual_risk": risk_data["inherent_risk"] * (1 - control["effectiveness"]),
                "cost": control["implementation_cost"] + control["annual_cost"],
                "feasibility": control["feasibility"],
                "effectiveness": control["effectiveness"]
            }
        
        # Transfer options (insurance, outsourcing)
        for transfer_option in risk_data.get("transfer_options", []):
            options[f"transfer_{transfer_option['type']}"] = {
                "residual_risk": transfer_option["retained_risk"],
                "cost": transfer_option["annual_cost"],
                "feasibility": transfer_option["availability"],
                "effectiveness": 1 - transfer_option["retained_risk_percentage"]
            }
        
        return options
    
    def calculate_risk_treatment_roi(self, treatment_option, risk_data):
        """Calculate return on investment for risk treatment"""
        # Risk reduction value
        risk_reduction = risk_data["inherent_risk"] - treatment_option["residual_risk"]
        
        # Calculate ROI
        roi = (risk_reduction - treatment_option["cost"]) / treatment_option["cost"]
        
        return {
            "risk_reduction_value": risk_reduction,
            "treatment_cost": treatment_option["cost"],
            "net_benefit": risk_reduction - treatment_option["cost"],
            "roi_percentage": roi * 100
        }
