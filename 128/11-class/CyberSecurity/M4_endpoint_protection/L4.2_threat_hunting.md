import pandas as pd
import numpy as np
from scipy import stats
from sklearn.cluster import DBSCAN
from sklearn.preprocessing import StandardScaler

class StatisticalThreatHunter:
    def __init__(self):
        self.baseline_stats = {}
        self.anomaly_detectors = {}
        
    def frequency_analysis_hunt(self, dns_logs):
        """
        DNS frequency analysis for C2 detection
        """
        # Calculate domain query frequencies
        domain_counts = dns_logs.groupby('domain').size().reset_index(name='query_count')
        
        # Identify statistical outliers
        mean_queries = domain_counts['query_count'].mean()
        std_queries = domain_counts['query_count'].std()
        
        # Find domains with unusual query patterns
        suspicious_domains = domain_counts[
            (domain_counts['query_count'] > mean_queries + 3 * std_queries) |
            (domain_counts['query_count'] < mean_queries - 2 * std_queries)
        ]
        
        # Analyze query timing patterns
        for domain in suspicious_domains['domain']:
            domain_logs = dns_logs[dns_logs['domain'] == domain]
            
            # Check for beaconing patterns
            time_diffs = domain_logs['timestamp'].diff().dt.total_seconds()
            
            # Detect regular intervals (potential C2 beaconing)
            if self.detect_beaconing_pattern(time_diffs):
                yield BeaconingDetection(
                    domain=domain,
                    query_count=len(domain_logs),
                    pattern_type="Regular Interval Beaconing",
                    confidence=self.calculate_beaconing_confidence(time_diffs),
                    evidence=domain_logs.to_dict('records')
                )
    
    def detect_beaconing_pattern(self, time_intervals):
        """
        Detect regular communication patterns
        """
        # Remove NaN values
        intervals = time_intervals.dropna()
        
        if len(intervals) < 5:
            return False
            
        # Calculate coefficient of variation
        cv = intervals.std() / intervals.mean()
        
        # Regular beaconing has low coefficient of variation
        return cv < 0.3 and len(intervals) > 10
    
    def behavioral_clustering_hunt(self, user_activity_logs):
        """
        User behavior clustering for insider threat detection
        """
        # Extract behavioral features
        user_features = self.extract_user_features(user_activity_logs)
        
        # Normalize features
        scaler = StandardScaler()
        normalized_features = scaler.fit_transform(user_features)
        
        # Perform clustering
        clustering = DBSCAN(eps=0.5, min_samples=5)
        clusters = clustering.fit_predict(normalized_features)
        
        # Identify outliers (cluster -1)
        outlier_indices = np.where(clusters == -1)[0]
        
        outlier_findings = []
        for idx in outlier_indices:
            user_id = user_features.index[idx]
            user_data = user_activity_logs[user_activity_logs['user_id'] == user_id]
            
            outlier_score = self.calculate_outlier_score(
                normalized_features[idx], 
                normalized_features[clusters != -1]
            )
            
            outlier_findings.append(OutlierUser(
                user_id=user_id,
                outlier_score=outlier_score,
                behavioral_anomalies=self.identify_anomalies(user_data),
                risk_level=self.assess_risk_level(outlier_score),
                investigation_priority=self.calculate_priority(user_data, outlier_score)
            ))
        
        return outlier_findings
    
    def process_injection_hunt(self, process_logs):
        """
        Hunt for process injection techniques
        """
        injection_indicators = []
        
        # Group by parent-child relationships
        process_trees = self.build_process_trees(process_logs)
        
        for tree in process_trees:
            # Check for suspicious parent-child relationships
            for process in tree.processes:
                
                # Unusual process spawning patterns
                if self.detect_unusual_spawning(process):
                    injection_indicators.append(ProcessAnomalyDetection(
                        type="Unusual Process Spawning",
                        process_id=process.pid,
                        parent_process=process.parent_name,
                        evidence=process.command_line,
                        confidence=0.7
                    ))
                
                # Memory manipulation indicators
                if self.detect_memory_manipulation(process):
                    injection_indicators.append(ProcessAnomalyDetection(
                        type="Memory Manipulation",
                        process_id=process.pid,
                        technique="Possible Process Injection",
                        evidence=process.memory_operations,
                        confidence=0.8
                    ))
        
        return injection_indicators
    
    def lateral_movement_hunt(self, network_logs, authentication_logs):
        """
        Hunt for lateral movement activities
        """
        # Correlate network and authentication events
        correlated_events = self.correlate_auth_network_events(
            network_logs, authentication_logs
        )
        
        lateral_movement_findings = []
        
        for user_session in correlated_events:
            # Check for rapid credential usage across multiple systems
            if self.detect_credential_stuffing(user_session):
                lateral_movement_findings.append(LateralMovementDetection(
                    technique="Credential Stuffing",
                    user_account=user_session.username,
                    affected_systems=user_session.target_systems,
                    time_window=user_session.duration,
                    confidence=0.8
                ))
            
            # Check for admin tool usage patterns
            if self.detect_admin_tool_abuse(user_session):
                lateral_movement_findings.append(LateralMovementDetection(
                    technique="Administrative Tool Abuse",
                    user_account=user_session.username,
                    tools_used=user_session.admin_tools,
                    privilege_escalations=user_session.privilege_changes,
                    confidence=0.9
                ))
        
        return lateral_movement_findings

        Assess current IOC quality and relevance
        """

            'usage_frequency': self.calculate_usage_frequency(ioc),
            'detection_rate': self.calculate_detection_rate(ioc),
            'false_positive_rate': self.calculate_fp_rate(ioc),
            'source_credibility': self.assess_source_credibility(ioc.source),
            'context_completeness': self.assess_context_completeness(ioc)
        }
        
        # Calculate weighted quality score
        weights = {
            'age': 0.15,
            'usage_frequency': 0.20,
            'detection_rate': 0.25,
            'false_positive_rate': 0.20,
            'source_credibility': 0.10,
            'context_completeness': 0.10
        }
        
        quality_score = sum(
            factor_value * weights[factor_name]
            for factor_name, factor_value in quality_factors.items()
        )
        
        return IOCQualityAssessment(

        """

            source_info = self.identify_log_source(log_entry)
            
            # Apply source-specific parsing
            parsed_event = self.parsers.parse_log(log_entry, source_info.parser_rules)

            normalized_event = self.normalizers.normalize_event(parsed_event, source_info)
            
            # Add enrichment data
            enriched_event = self.enrich_event(normalized_event)
            
            # Validate data quality

                normalized_events.append(enriched_event)

    def __init__(self):
        self.event_queue = EventQueue()
        self.correlation_engine = CorrelationEngine()
        self.alert_manager = AlertManager()
        self.threat_detector = ThreatDetectionEngine()
        
    def process_event_stream(self, event_stream):
        """
        Process real-time event stream
        """
        processed_events = []
        
        for event in event_stream:
            # Stage 1: Event preprocessing
            preprocessed_event = self.preprocess_event(event)
            
            # Stage 2: Real-time correlation
            correlation_results = self.correlation_engine.correlate_event(preprocessed_event)
            
            # Stage 3: Threat detection
            threat_analysis = self.threat_detector.analyze_event(preprocessed_event, correlation_results)
            
            # Stage 4: Alert generation
            if threat_analysis.threat_level > 0.5:
                alert = self.alert_manager.generate_alert(
                    preprocessed_event, correlation_results, threat_analysis
                )
                self.alert_manager.dispatch_alert(alert)
            
            # Stage 5: Event storage
            enriched_event = EnrichedSecurityEvent(
                original_event=preprocessed_event,
                correlation_data=correlation_results,
                threat_analysis=threat_analysis,
                processing_metadata=self.get_processing_metadata()
            )
            
            processed_events.append(enriched_event)
        
        return processed_events
    
    def preprocess_event(self, raw_event):

        Execute specific correlation rule against events

        candidate_events = self.filter_events_by_conditions(events, rule.conditions)
        
        if not candidate_events:

        # Apply correlation logic
        if rule.correlation_logic == 'temporal_and_spatial':
            correlations = self.temporal_spatial_correlation(candidate_events)
        elif rule.correlation_logic == 'temporal_and_behavioral':
            correlations = self.temporal_behavioral_correlation(candidate_events)
        elif rule.correlation_logic == 'behavioral_and_statistical':
            correlations = self.behavioral_statistical_correlation(candidate_events)
        else:
            correlations = self.generic_correlation(candidate_events, rule.correlation_logic)
        
        # Calculate confidence scores
        for correlation in correlations:
            confidence_score = self.calculate_correlation_confidence(correlation, rule)
            
            if confidence_score >= rule.confidence_threshold:
                rule_execution.add_positive_correlation(correlation, confidence_score)
        
        return rule_execution

                    tuning_results.add_recommendation(

                        'remove_condition',

                    )

class SOARWorkflowEngine:
    def __init__(self):
        self.task_scheduler = TaskScheduler()
        self.execution_monitor = ExecutionMonitor()
        self.error_handler = ErrorHandler()
        self.state_manager = StateManager()
        
    def execute_playbook(self, playbook, incident_data):
        """
        Execute SOAR playbook with advanced workflow management
        """
        execution_context = PlaybookExecutionContext(
            playbook=playbook,
            incident_data=incident_data,
            execution_id=self.generate_execution_id(),
            start_time=datetime.utcnow()
        )
        
        try:
            # Initialize execution state
            self.state_manager.initialize_execution_state(execution_context)
            
            # Execute playbook phases
            for phase in playbook.phases:
                phase_result = self.execute_phase(phase, execution_context)
                execution_context.add_phase_result(phase_result)
                
                # Check for early termination conditions
                if phase_result.termination_requested:
                    break
            
            # Finalize execution
            execution_result = self.finalize_execution(execution_context)
            
        except Exception as e:
            # Handle execution errors
            execution_result = self.error_handler.handle_execution_error(
                e, execution_context
            )
        
        return execution_result
    
    def execute_phase(self, phase, execution_context):
        """
        Execute individual playbook phase
        """
        phase_execution = PhaseExecution(phase, execution_context)
        
        # Prepare tasks for execution
        ready_tasks = self.identify_ready_tasks(phase.tasks)
        
        while ready_tasks or phase_execution.has_running_tasks():
            # Start new tasks
            for task in ready_tasks:
                if self.can_start_task(task, phase_execution):
                    task_execution = self.start_task_execution(task, execution_context)
                    phase_execution.add_running_task(task_execution)
            
            # Monitor running tasks
            completed_tasks = self.monitor_task_completion(phase_execution.running_tasks)
            
            for completed_task in completed_tasks:
                phase_execution.mark_task_completed(completed_task)
                
                # Update execution context with task results
                execution_context.update_context(completed_task.results)
                
                # Check for newly ready tasks
                newly_ready = self.check_dependencies_satisfied(
                    phase.tasks, phase_execution.completed_tasks
                )
                ready_tasks.extend(newly_ready)
            
            # Process decisions
            for decision in phase.decisions:
                if self.can_evaluate_decision(decision, execution_context):
                    decision_result = self.evaluate_decision(decision, execution_context)
                    phase_execution.add_decision_result(decision_result)
                    
                    # Apply decision outcome
                    self.apply_decision_outcome(decision_result, phase_execution)
            
            # Brief sleep to prevent busy waiting
            time.sleep(0.1)
        
        return phase_execution.get_phase_result()
    
    def implement_dynamic_playbook_adaptation(self, execution_context):
        """
        Implement dynamic playbook adaptation based on runtime conditions
        """
        adaptation_engine = PlaybookAdaptationEngine()
        
        # Analyze current execution state
        current_state = execution_context.get_current_state()
        
        # Identify adaptation opportunities
        adaptation_opportunities = adaptation_engine.identify_adaptations(
            current_state, execution_context.incident_data
        )
        
        adaptations_applied = []
        
        for opportunity in adaptation_opportunities:
            if opportunity.type == "skip_task":
                # Skip unnecessary tasks based on conditions
                if self.evaluate_skip_condition(opportunity.condition, execution_context):
                    self.skip_task(opportunity.target_task, execution_context)
                    adaptations_applied.append(opportunity)
            
            elif opportunity.type == "add_task":
                # Add additional tasks based on findings
                if self.evaluate_add_condition(opportunity.condition, execution_context):
                    self.add_dynamic_task(opportunity.new_task, execution_context)
                    adaptations_applied.append(opportunity)

### Основни възможности на EPP (Endpoint Protection Platform)

**Подход, ориентиран към превенция:**

- **Device Control** - управление на достъпа до USB и периферни устройства
- **Web Protection** - блокиране на злонамерени URL-и и превенция на phishing
- **Firewall Integration** - управление на host-базирана защитна стена
- **Vulnerability Assessment** - сканиране и кръпване на endpoint уязвимости

class ProcessBehaviorAnalyzer:
    def __init__(self):
        self.baseline_behaviors = {}
        self.anomaly_threshold = 2.5  # Standard deviations
        self.ml_model = load_pretrained_model('process_behavior_v2.model')

        Analyze process behavior using machine learning
        """
        behavior_features = self.extract_behavior_features(process_data)
