import numpy as np

class EndpointAnomalyDetector:

            contamination=0.1,  # Expect 10% anomalies

            n_estimators=200

        self.dbscan = DBSCAN(eps=0.5, min_samples=5)
        self.scaler = StandardScaler()
        self.baseline_models = {}
        
    def train_baseline_models(self, historical_data):
        """
        Train baseline models using historical endpoint data
        """
        # Process different data types
        for endpoint_type in ['workstation', 'server', 'mobile']:
            type_data = historical_data[endpoint_type]
            
            # Feature extraction
            features = self.extract_behavioral_features(type_data)
            
            # Normalize features
            normalized_features = self.scaler.fit_transform(features)
            
            # Train isolation forest for this endpoint type
            if_model = IsolationForest(contamination=0.05)
            if_model.fit(normalized_features)
            
            # Train clustering model
            clusters = self.dbscan.fit_predict(normalized_features)
            
            self.baseline_models[endpoint_type] = {
                'isolation_forest': if_model,
                'clusters': clusters,
                'feature_means': np.mean(normalized_features, axis=0),
                'feature_stds': np.std(normalized_features, axis=0)
            }
    
    def detect_anomalies(self, endpoint_data):
        """
        Detect anomalies in real-time endpoint data
        """
        endpoint_type = endpoint_data.get('type', 'workstation')

        if_score = model['isolation_forest'].decision_function(normalized_features)[0]

# Deep Learning Threat Detection Model
import tensorflow as tf
from tensorflow.keras.models import Sequential, Model
from tensorflow.keras.layers import LSTM, Dense, Dropout, Attention

class DeepThreatDetectionModel:
    def __init__(self):
        self.sequence_length = 100  # Time series length
        self.feature_dim = 50       # Feature dimensions
        self.model = self.build_model()
    
    def build_model(self):
        """
        Build deep learning model for threat detection
        """
        # Input layer
        inputs = tf.keras.Input(shape=(self.sequence_length, self.feature_dim))
        
        # LSTM layers for temporal pattern recognition
        lstm1 = LSTM(128, return_sequences=True)(inputs)
        lstm1_dropout = Dropout(0.3)(lstm1)
        
        lstm2 = LSTM(64, return_sequences=True)(lstm1_dropout)
        lstm2_dropout = Dropout(0.3)(lstm2)
        
        # Attention mechanism for important feature focus
        attention = Attention()([lstm2_dropout, lstm2_dropout])
        
        # Dense layers for classification
        dense1 = Dense(32, activation='relu')(attention)
        dense1_dropout = Dropout(0.5)(dense1)
        
        # Output layers
        malware_output = Dense(1, activation='sigmoid', name='malware_detection')(dense1_dropout)
        attack_type_output = Dense(10, activation='softmax', name='attack_classification')(dense1_dropout)
        severity_output = Dense(1, activation='linear', name='severity_prediction')(dense1_dropout)
        
        model = Model(
            inputs=inputs,
            outputs=[malware_output, attack_type_output, severity_output]
        )
        
        # Compile с multiple loss functions
        model.compile(
            optimizer='adam',
            loss={
                'malware_detection': 'binary_crossentropy',
                'attack_classification': 'categorical_crossentropy',
                'severity_prediction': 'mse'
            },
            loss_weights={
                'malware_detection': 1.0,
                'attack_classification': 0.8,
                'severity_prediction': 0.5
            },
            metrics=['accuracy']
        )
        
        return model
    
    def preprocess_endpoint_data(self, endpoint_telemetry):
        """
        Preprocess endpoint telemetry for deep learning
        """
        # Convert telemetry to time series format
        time_series_data = []
        
        for timestamp in sorted(endpoint_telemetry.keys()):
            data_point = endpoint_telemetry[timestamp]
            
            feature_vector = [
                # Process features
                data_point.get('process_creations', 0),
                data_point.get('process_terminations', 0),
                data_point.get('process_injections', 0),
                
                # Network features
                data_point.get('network_connections', 0),
                data_point.get('dns_queries', 0),
                data_point.get('http_requests', 0),
                
                # File system features
                data_point.get('file_creations', 0),
                data_point.get('file_modifications', 0),
                data_point.get('file_deletions', 0),
                
                # Registry features
                data_point.get('registry_reads', 0),
                data_point.get('registry_writes', 0),
                data_point.get('registry_deletions', 0),
                
                # Memory features
                data_point.get('memory_allocations', 0),
                data_point.get('memory_protections', 0),
                data_point.get('dll_loads', 0),
                
                # Additional behavioral features...
            ]
            
            # Pad vector to feature_dim length
            while len(feature_vector) < self.feature_dim:
                feature_vector.append(0.0)
            
            time_series_data.append(feature_vector[:self.feature_dim])
        
        # Ensure sequence length
        if len(time_series_data) < self.sequence_length:
            # Pad sequence
            padding_needed = self.sequence_length - len(time_series_data)
            time_series_data = [[0.0] * self.feature_dim] * padding_needed + time_series_data
        elif len(time_series_data) > self.sequence_length:
            # Take latest sequence_length data points
            time_series_data = time_series_data[-self.sequence_length:]
        
        return np.array([time_series_data])  # Batch dimension
    
    def predict_threats(self, endpoint_telemetry):
        """
        Predict threats using trained deep learning model
        """
        preprocessed_data = self.preprocess_endpoint_data(endpoint_telemetry)
        
        # Get predictions
        predictions = self.model.predict(preprocessed_data)
        malware_prob, attack_types, severity = predictions
        
        # Process results
        is_malware = malware_prob[0][0] > 0.5
        attack_type_idx = np.argmax(attack_types[0])
        severity_score = severity[0][0]
        
        attack_types_mapping = [
            'Ransomware', 'Banking Trojan', 'Spyware', 'Adware',
            'Rootkit', 'Worm', 'APT', 'Cryptominer', 'Botnet', 'Other'
        ]
        
        return ThreatPrediction(
            is_malware=is_malware,
            malware_confidence=float(malware_prob[0][0]),
            attack_type=attack_types_mapping[attack_type_idx],
            attack_confidence=float(attack_types[0][attack_type_idx]),
            severity_score=float(severity_score),
            recommendations=self.generate_response_recommendations(
                is_malware, attack_type_idx, severity_score
            )
        )
