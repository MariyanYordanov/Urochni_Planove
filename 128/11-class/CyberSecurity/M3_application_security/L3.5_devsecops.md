**GitHub Actions DevSecOps Pipeline:**
```yaml
name: DevSecOps Security Pipeline
on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  # Security scanning jobs
  secret-scanning:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
      with:
        fetch-depth: 0

    - name: Run TruffleHog Secret Scanner
      uses: trufflesecurity/trufflehog@main
      with:
        path: ./
        base: main
        head: HEAD
        extra_args: --debug --only-verified

    - name: Run GitLeaks Secret Detection
      uses: gitleaks/gitleaks-action@v3
      env:
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

  sast-analysis:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Set up JDK 17
      uses: actions/setup-java@v3
      with:
        java-version: '17'
        distribution: 'temurin'

    - name: Run SonarQube Analysis
      uses: SonarSource/sonarcloud-github-action@master
      env:
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
      with:
        scanMetadataReportFile: target/sonar/report-task.txt

    - name: Run Semgrep SAST
      uses: returntocorp/semgrep-action@v1
      with:
        config: auto
        generateSarif: "1"

    - name: Upload SARIF results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: semgrep.sarif

  dependency-check:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Run OWASP Dependency Check
      uses: dependency-check/Dependency-Check_Action@main
      env:
        JAVA_HOME: /opt/jdk
      with:
        project: 'secure-app'
        path: '.'
        format: 'ALL'
        args: >
          --enableRetired
          --enableExperimental
          --failOnCVSS 7

    - name: Upload Dependency Check Results
      uses: actions/upload-artifact@v3
      with:
        name: dependency-check-report
        path: reports/

  build-and-test:
    needs: [secret-scanning, sast-analysis, dependency-check]
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Set up JDK 17
      uses: actions/setup-java@v3
      with:
        java-version: '17'
        distribution: 'temurin'

    - name: Build Application
      run: |
        ./mvnw clean compile
        ./mvnw package -DskipTests

    - name: Run Security Unit Tests
      run: ./mvnw test -Dtest=**/*SecurityTest

    - name: Build Docker Image
      run: |
        docker build -t secure-app:${{ github.sha }} .
        docker tag secure-app:${{ github.sha }} secure-app:latest

  container-security:
    needs: build-and-test
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Run Trivy Container Scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'secure-app:latest'
        format: 'sarif'
        output: 'trivy-results.sarif'

    - name: Run Anchore Container Scan
      uses: anchore/scan-action@v3
      with:
        image: "secure-app:latest"
        fail-build: true
        severity-cutoff: high

    - name: Upload Trivy Results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: 'trivy-results.sarif'

  dast-testing:
    needs: build-and-test
    runs-on: ubuntu-latest
    services:
      app:
        image: secure-app:latest
        ports:
          - 8080:8080
    steps:
    - name: Wait for Application Startup
      run: |
        timeout 60 bash -c 'until curl -f http://localhost:8080/actuator/health; do sleep 2; done'

    - name: Run OWASP ZAP Baseline Scan
      uses: zaproxy/action-baseline@v0.7.0
      with:
        target: 'http://localhost:8080'
        rules_file_name: '.zap/rules.tsv'
        cmd_options: '-a'

    - name: Run OWASP ZAP Full Scan
      uses: zaproxy/action-full-scan@v0.4.0
      with:
        target: 'http://localhost:8080'
        rules_file_name: '.zap/rules.tsv'
        cmd_options: '-a -j'

  infrastructure-security:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3

    - name: Run Terraform Security Scan
      uses: aquasecurity/tfsec-action@v1.0.0
      with:
        soft_fail: false

    - name: Run Checkov IaC Scanner
      uses: bridgecrewio/checkov-action@master
      with:
        directory: ./infrastructure
        framework: terraform,kubernetes,dockerfile
        output_format: sarif
        output_file_path: checkov-results.sarif

    - name: Upload Checkov Results
      uses: github/codeql-action/upload-sarif@v2
      with:
        sarif_file: checkov-results.sarif

  security-gate:
    needs: [container-security, dast-testing, infrastructure-security]
    runs-on: ubuntu-latest
    steps:
    - name: Security Quality Gate
      run: |
        # Aggregate security results и determine if deployment should proceed
        echo "Evaluating security quality gate..."

        # Check for critical vulnerabilities
        CRITICAL_VULNS=$(cat security-results.json | jq '.critical_count')
        HIGH_VULNS=$(cat security-results.json | jq '.high_count')

        if [ "$CRITICAL_VULNS" -gt 0 ]; then
          echo " Security gate failed: $CRITICAL_VULNS critical vulnerabilities found"
          exit 1
        fi

        if [ "$HIGH_VULNS" -gt 5 ]; then
          echo " Security gate failed: $HIGH_VULNS high-severity vulnerabilities (max 5 allowed)"
          exit 1
        fi

        echo " Security gate passed: No critical vulnerabilities found"

  deploy:
    needs: security-gate
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
    - name: Deploy to Production
      run: |
        echo "Deploying secure application to production..."
        # Deployment logic here
```

### **Интеграция на инструменти за автоматизация на сигурността (12 мин)**

**Jenkins Security Pipeline:**
```groovy
pipeline {
    agent any

    environment {
        DOCKER_REGISTRY = 'secure-registry.company.com'
        SONAR_TOKEN = credentials('sonar-token')
        SECURITY_GATEWAY_URL = 'https://security-gateway.company.com'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
                script {
                    env.GIT_COMMIT_SHORT = sh(
                        script: "git rev-parse --short HEAD",
                        returnStdout: true
                    ).trim()
                }
            }
        }

        stage('Security Pre-Flight') {
            parallel {
                stage('Secret Scanning') {
                    steps {
                        script {
                            def secretScanResults = sh(
                                script: 'trufflehog filesystem . --json',
                                returnStdout: true
                            )

                            if (secretScanResults.contains('"verified": true')) {
                                error " Verified secrets found в repository!"
                            }
                        }
                    }
                }

                stage('License Compliance') {
                    steps {
                        sh '''
                            # FOSSA license scanning
                            fossa analyze
                            fossa test --timeout 300
                        '''
                    }
                }

                stage('Branch Protection Validation') {
                    steps {
                        script {
                            if (env.BRANCH_NAME == 'main') {
                                def prNumber = env.CHANGE_ID
                                if (!prNumber) {
                                    error " Direct push to main branch не е allowed!"
                                }
                            }
                        }
                    }
                }
            }
        }

        stage('Build & SAST Analysis') {
            parallel {
                stage('Application Build') {
                    steps {
                        sh '''
                            ./mvnw clean compile
                            ./mvnw package -DskipTests
                        '''
                    }
                }

                stage('Static Security Analysis') {
                    steps {
                        // SonarQube security analysis
                        withSonarQubeEnv('SonarQube') {
                            sh '''
                                ./mvnw sonar:sonar \
                                  -Dsonar.projectKey=secure-app \
                                  -Dsonar.qualitygate.wait=true \
                                  -Dsonar.security-hotspots.analysis=true
                            '''
                        }

                        // CodeQL analysis
                        sh '''
                            codeql database create codeql-db --language=java
                            codeql database analyze codeql-db \
                                --format=sarif-latest \
                                --output=security-results.sarif \
                                --sarif-category=security
                        '''

                        // Semgrep analysis
                        sh '''
                            semgrep --config=auto \
                                --sarif \
                                --output=semgrep-results.sarif \
                                --severity=ERROR \
                                .
                        '''
                    }
                }

                stage('Dependency Vulnerability Scan') {
                    steps {
                        sh '''
                            # OWASP Dependency Check
                            dependency-check.sh \
                                --project "secure-app" \
                                --scan . \
                                --format ALL \
                                --failOnCVSS 7 \
                                --enableExperimental

                            # Snyk dependency scanning
                            snyk test --severity-threshold=high
                            snyk monitor
                        '''
                    }
                }
            }
        }

        stage('Security Quality Gate') {
            steps {
                script {
                    // Aggregate security findings
                    def securityReport = [:]

                    // Parse SonarQube results
                    def sonarResults = readJSON file: 'sonar-results.json'
                    securityReport.sonar = [
                        securityHotspots: sonarResults.issues.findAll { it.type == 'SECURITY_HOTSPOT' }.size(),
                        vulnerabilities: sonarResults.issues.findAll { it.type == 'VULNERABILITY' }.size()
                    ]

                    // Security gate decision logic
                    if (securityReport.dependencies.critical > 0) {
                        error " Security Gate Failed: ${securityReport.dependencies.critical} critical vulnerabilities в dependencies"
                    }

                    if (securityReport.sonar.vulnerabilities > 0) {
                        error " Security Gate Failed: ${securityReport.sonar.vulnerabilities} code vulnerabilities found"
                    }

                    echo " Security Gate Passed: Application meets security requirements"
                }
            }
        }

        stage('Container Security') {
            when {
                expression { currentBuild.result != 'FAILURE' }
            }
            steps {
                script {
                    def image = docker.build("${DOCKER_REGISTRY}/secure-app:${env.GIT_COMMIT_SHORT}")

                    sh """
                        # Trivy container scan
                        trivy image \
                            --severity CRITICAL,HIGH \
                            --format json \
                            --output trivy-results.json \
                            ${DOCKER_REGISTRY}/secure-app:${env.GIT_COMMIT_SHORT}
                    """

                    image.push()
                    image.push('latest')
                }
            }
        }

        stage('Infrastructure Security Validation') {
            steps {
                sh '''
                    # Terraform security validation
                    tfsec ./infrastructure --format json --out tfsec-results.json

                    # Checkov Infrastructure as Code scanning
                    checkov -d ./infrastructure \
                        --framework terraform,kubernetes \
                        --output json \
                        --output-file checkov-results.json

                    # OPA policy validation
                    opa test ./policies/ --verbose
                '''
            }
        }
    }

    post {
        always {
            archiveArtifacts artifacts: '*-results.json,*-results.sarif', allowEmptyArchive: true
        }

        success {
            echo " DevSecOps Pipeline completed successfully!"
        }

        failure {
            echo " DevSecOps Pipeline failed due to security issues"
        }
    }
}
```

### **Security метрики и непрекъснат мониторинг (13 мин)**

**DevSecOps Metrics Dashboard — ключови концепции:**

Ключови security KPI за проследяване в DevSecOps:
- **Mean Time to Detection (MTTD)** — времето от въвеждане на уязвимост до нейното засичане
- **Mean Time to Remediation (MTTR)** — времето от засичане до отстраняване
- **Security debt** — натрупани неотстранени security findings
- **Pipeline security coverage** — процент от пътеките на кода, тествани за сигурност
- **Compliance score** — съответствие с security стандарти

**Категории ключови метрики:**
```
Pipeline Metrics:
- Security scan pass/fail rates
- Vulnerability counts by severity (Critical/High/Medium/Low)
- Time to remediation by severity
- Security gate failure rates

Code Quality Metrics:
- Security hotspots identified
- Code vulnerabilities found
- Dependency vulnerabilities
- Secret exposure incidents

Deployment Metrics:
- Security-validated deployments
- Deployment frequency
- Change failure rate (security-related)
- Recovery time from security incidents
```

**Pipeline за Terraform Security Validation:**
```yaml
name: Infrastructure Security Validation
on:
  push:
    paths:
      - 'infrastructure/**'
  pull_request:
    paths:
      - 'infrastructure/**'

jobs:
  terraform-security:
    runs-on: ubuntu-latest

    - name: Run tfsec Security Scan
      uses: aquasecurity/tfsec-action@v1.0.0
      with:
        soft_fail: false
        tfsec_args: --minimum-severity MEDIUM

    - name: Run Checkov IaC Security Scan
      uses: bridgecrewio/checkov-action@master
      with:
        directory: infrastructure/
        framework: terraform
        output_format: sarif
        output_file_path: checkov-results.sarif

    - name: Security Gate Evaluation
      run: |
        CRITICAL_ISSUES=$(jq '.critical_count' security-summary.json)

        if [ "$CRITICAL_ISSUES" -gt 0 ]; then

        fi
```
