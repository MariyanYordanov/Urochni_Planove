```java
// Secure API Controller с Object-Level Authorization
@RestController
@RequestMapping("/api")
@PreAuthorize("hasRole('USER')")
public class SecureOrderController {

    @Autowired
    private OrderService orderService;

    @Autowired
    private AuthorizationService authService;

    @GetMapping("/users/{userId}/orders/{orderId}")
    public ResponseEntity<OrderResponse> getUserOrder(
            @PathVariable Long userId,
            @PathVariable Long orderId,
            Authentication authentication) {

        // Step 1: Verify user identity
        String currentUsername = authentication.getName();
        if (!authService.isUserAuthorized(currentUsername, userId)) {
            throw new AccessDeniedException("Cannot access other users' data");
        }

        // Step 2: Verify object ownership
        Order order = orderService.findById(orderId);
        if (order == null) {
            throw new ResourceNotFoundException("Order not found");
        }

        if (!order.getUserId().equals(userId)) {
            throw new AccessDeniedException("Order does not belong to user");
        }

        // Step 3: Apply data filtering based on permissions
        OrderResponse response = authService.filterOrderData(order, authentication);

        // Step 4: Log access attempt
        auditService.logDataAccess(currentUsername, "ORDER_ACCESS", orderId);

        return ResponseEntity.ok(response);
    }

    // Bulk operations требват additional security
    @GetMapping("/users/{userId}/orders")
    @PreAuthorize("@authService.canAccessUserOrders(#userId, authentication)")
    public ResponseEntity<List<OrderResponse>> getUserOrders(
            @PathVariable Long userId,
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "10") int size,
            Authentication authentication) {

        // Implement pagination limits to prevent data dumping
        if (size > 100) {
            throw new IllegalArgumentException("Page size cannot exceed 100");
        }

        Pageable pageable = PageRequest.of(page, size);
        Page<Order> orders = orderService.findByUserId(userId, pageable);

        List<OrderResponse> responses = orders.getContent().stream()
            .map(order -> authService.filterOrderData(order, authentication))
            .collect(Collectors.toList());

        return ResponseEntity.ok(responses);
    }
}
```

```java
// Advanced Rate Limiting Implementation
@Component
public class APIRateLimitingFilter implements Filter {

    private final RedisTemplate<String, String> redisTemplate;
    private final RateLimitingConfig config;

    @Override
    public void doFilter(ServletRequest request, ServletResponse response,
                        FilterChain chain) throws IOException, ServletException {

        HttpServletRequest httpRequest = (HttpServletRequest) request;
        HttpServletResponse httpResponse = (HttpServletResponse) response;

        String clientId = extractClientIdentifier(httpRequest);
        String endpoint = httpRequest.getRequestURI();

        // Apply endpoint-specific rate limits
        RateLimitRule rule = config.getRuleForEndpoint(endpoint);

        if (!checkRateLimit(clientId, endpoint, rule)) {
            httpResponse.setStatus(429); // Too Many Requests
            httpResponse.setHeader("Retry-After", String.valueOf(rule.getWindowSeconds()));
            httpResponse.setHeader("X-RateLimit-Limit", String.valueOf(rule.getRequestsPerWindow()));
            httpResponse.setHeader("X-RateLimit-Remaining", "0");

            // Return rate limit exceeded response
            ObjectMapper mapper = new ObjectMapper();
            ErrorResponse error = new ErrorResponse(
                "RATE_LIMIT_EXCEEDED",
                "API rate limit exceeded. Please retry after " + rule.getWindowSeconds() + " seconds"
            );
            httpResponse.getWriter().write(mapper.writeValueAsString(error));
            return;
        }

        chain.doFilter(request, response);
    }

    private boolean checkRateLimit(String clientId, String endpoint, RateLimitRule rule) {
        String key = String.format("rate_limit:%s:%s", clientId, endpoint);
        long currentWindow = System.currentTimeMillis() / (rule.getWindowSeconds() * 1000);
        String windowKey = key + ":" + currentWindow;

        // Sliding window rate limiting using Redis
        Long currentCount = redisTemplate.opsForValue().increment(windowKey);

        if (currentCount == 1) {
            // Set expiration for the window
            redisTemplate.expire(windowKey, Duration.ofSeconds(rule.getWindowSeconds()));
        }

        return currentCount <= rule.getRequestsPerWindow();
    }

    private String extractClientIdentifier(HttpServletRequest request) {
        // Priority: API Key > JWT Subject > IP Address
        String apiKey = request.getHeader("X-API-Key");
        if (apiKey != null) {
            return "api_key:" + apiKey;
        }

        String authHeader = request.getHeader("Authorization");
        if (authHeader != null && authHeader.startsWith("Bearer ")) {
            try {
                String token = authHeader.substring(7);
                Claims claims = jwtService.validateToken(token);
                return "user:" + claims.getSubject();
            } catch (Exception e) {
                // Fall back to IP-based limiting
            }
        }

        return "ip:" + getClientIpAddress(request);
    }
}
```

```java
// API Request Validation Framework
@Component
public class APIValidationService {

    // JSON Schema Validation
    public void validateJsonPayload(String jsonPayload, String schemaName) {
        try {
            JsonSchema schema = loadJsonSchema(schemaName);
            JsonNode jsonNode = objectMapper.readTree(jsonPayload);

            Set<ValidationMessage> violations = schema.validate(jsonNode);
            if (!violations.isEmpty()) {
                throw new ValidationException(
                    "JSON payload validation failed: " + violations.toString()
                );
            }
        } catch (Exception e) {
            throw new ValidationException("Invalid JSON payload", e);
        }
    }

    // Content Type Validation
    public void validateContentType(String contentType, List<String> allowedTypes) {
        if (contentType == null || !allowedTypes.contains(contentType.toLowerCase())) {
            throw new UnsupportedMediaTypeException(
                "Content type must be one of: " + String.join(", ", allowedTypes)
            );
        }
    }

    // Request Size Validation
    public void validateRequestSize(long contentLength, long maxSize) {
        if (contentLength > maxSize) {
            throw new PayloadTooLargeException(
                String.format("Request size %d exceeds maximum allowed size %d",
                             contentLength, maxSize)
            );
        }
    }

    // Parameter Injection Prevention
    public Map<String, Object> sanitizeParameters(Map<String, Object> parameters) {
        Map<String, Object> sanitized = new HashMap<>();

        for (Map.Entry<String, Object> entry : parameters.entrySet()) {
            String key = entry.getKey();
            Object value = entry.getValue();

            // Validate parameter names (prevent parameter pollution)
            if (!isValidParameterName(key)) {
                throw new ValidationException("Invalid parameter name: " + key);
            }

            // Sanitize parameter values based on expected type
            Object sanitizedValue = sanitizeParameterValue(value);
            sanitized.put(key, sanitizedValue);
        }

        return sanitized;
    }

    private boolean isValidParameterName(String name) {
        // Allow only alphanumeric characters и underscores
        return name.matches("^[a-zA-Z0-9_]+$") && name.length() <= 50;
    }

    private Object sanitizeParameterValue(Object value) {
        if (value instanceof String) {
            String strValue = (String) value;

            // Remove potentially dangerous characters
            strValue = strValue.replaceAll("[<>\"'&]", "");

            // Limit string length
            if (strValue.length() > 1000) {
                strValue = strValue.substring(0, 1000);
            }

            return strValue;
        }

        return value; // Numbers, booleans, etc. pass through
    }
}
```

```java
        httpResponse.setHeader("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS");

            "Authorization, Content-Type, X-API-Key, X-Requested-With");
        httpResponse.setHeader("Access-Control-Max-Age", "86400");

        // Security policy headers
        httpResponse.setHeader("Strict-Transport-Security",
            "max-age=31536000; includeSubDomains; preload");

        chain.doFilter(request, response);
    }
```

```java
    // Generic error response structure
    @ExceptionHandler(Exception.class)
    public ResponseEntity<ErrorResponse> handleGenericException(Exception e,
                                                              HttpServletRequest request) {
        // Log detailed error untuk debugging (не expose to client)
        String errorId = UUID.randomUUID().toString();
        logger.error("API Error [{}]: {} at {}", errorId, e.getMessage(),

        // Return generic error message
        ErrorResponse response = new ErrorResponse(
            "INTERNAL_SERVER_ERROR",
            "An internal error occurred. Please contact support with error ID: " + errorId,

        );

        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(response);
    }

    // Validation errors (safe to expose)
    @ExceptionHandler(ValidationException.class)
    public ResponseEntity<ErrorResponse> handleValidationException(ValidationException e) {
        ErrorResponse response = new ErrorResponse(
            "VALIDATION_ERROR",
            "Request validation failed",
            sanitizeValidationErrors(e.getErrors())
        );

        return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(response);
    }

    // Authentication errors
    @ExceptionHandler(AuthenticationException.class)
    public ResponseEntity<ErrorResponse> handleAuthenticationException(AuthenticationException e) {
        // Don't reveal authentication details
        ErrorResponse response = new ErrorResponse(

        );

    }

    // Authorization errors
    @ExceptionHandler(AccessDeniedException.class)
    public ResponseEntity<ErrorResponse> handleAccessDeniedException(AccessDeniedException e) {
        // Don't reveal what resources exist
        ErrorResponse response = new ErrorResponse(
            "ACCESS_DENIED",
            "You don't have permission to access this resource"
        );

        return ResponseEntity.status(HttpStatus.FORBIDDEN).body(response);
    }

    // Resource not found
    @ExceptionHandler(ResourceNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleResourceNotFoundException(ResourceNotFoundException e) {
        ErrorResponse response = new ErrorResponse(
            "RESOURCE_NOT_FOUND",
            "The requested resource was not found"
        );

        return ResponseEntity.status(HttpStatus.NOT_FOUND).body(response);
    }

    // Rate limiting errors
    @ExceptionHandler(RateLimitExceededException.class)
    public ResponseEntity<ErrorResponse> handleRateLimitException(RateLimitExceededException e) {
        ErrorResponse response = new ErrorResponse(
            "RATE_LIMIT_EXCEEDED",
            "API rate limit exceeded. Please retry after the specified time.",
            Map.of("retryAfter", e.getRetryAfterSeconds())
        );

        HttpHeaders headers = new HttpHeaders();
        headers.add("Retry-After", String.valueOf(e.getRetryAfterSeconds()));

        return ResponseEntity.status(HttpStatus.TOO_MANY_REQUESTS)
            .headers(headers)
            .body(response);

    }
```

```java
                .timestamp(Instant.now())
                .username(username)
                .method(request.getMethod())
                .endpoint(request.getRequestURI())
                .queryParams(sanitizeQueryParams(request.getParameterMap()))
                .userAgent(request.getHeader("User-Agent"))
                .clientIp(getClientIpAddress(request))
                .responseStatus(response.getStatus())
                .processingTimeMs(processingTime)
                .requestId(MDC.get("requestId"))
                .build();

            auditLogger.info(objectMapper.writeValueAsString(event));

            // Send to SIEM или security monitoring system
            securityMonitoringService.processAuditEvent(event);

        } catch (Exception e) {
            // Don't fail the request if logging fails
            logger.error("Failed to log API audit event", e);
        }
    }

    private Map<String, String> sanitizeQueryParams(Map<String, String[]> parameterMap) {
        Map<String, String> sanitized = new HashMap<>();

        for (Map.Entry<String, String[]> entry : parameterMap.entrySet()) {
            String key = entry.getKey();
            String[] values = entry.getValue();

            // Don't log sensitive parameters
            if (isSensitiveParameter(key)) {
                sanitized.put(key, "[REDACTED]");
            } else {
                sanitized.put(key, String.join(",", values));
            }
        }

        return sanitized;
    }

    private boolean isSensitiveParameter(String paramName) {
        List<String> sensitiveParams = Arrays.asList(
            "password", "token", "secret", "key", "auth", "credential"
        );

        return sensitiveParams.stream()
            .anyMatch(sensitive -> paramName.toLowerCase().contains(sensitive));
    }
}
```

```java
        return tokenServices;
    }

    @Bean
    public TokenStore tokenStore() {
        // Use Redis за distributed token storage
        return new RedisTokenStore(jedisConnectionFactory());
    }

    @Bean
    public ClientDetailsService clientDetailsService() {
        return new JdbcClientDetailsService(dataSource()) {
            @Override
            public ClientDetails loadClientByClientId(String clientId)
                    throws ClientRegistrationException {

                ClientDetails client = super.loadClientByClientId(clientId);

                // Additional security validations
                validateClientSecurity(client);

                return client;
            }
        };
    }

    @Override
    public void configure(AuthorizationServerEndpointsConfigurer endpoints) {
        endpoints
            .tokenStore(tokenStore())
            .tokenServices(tokenServices())
            .userDetailsService(userDetailsService)
            .authenticationManager(authenticationManager)
            .tokenEnhancer(tokenEnhancerChain())
            .exceptionTranslator(oauth2ResponseExceptionTranslator());
    }

    @Override
    public void configure(AuthorizationServerSecurityConfigurer security) {
        security
            .tokenKeyAccess("hasAuthority('ROLE_TRUSTED_CLIENT')")
            .checkTokenAccess("hasAuthority('ROLE_TRUSTED_CLIENT')")
            .allowFormAuthenticationForClients()
            .passwordEncoder(passwordEncoder)
            .sslOnly() // Require HTTPS
            .realm("oauth2/client");
    }

    private void validateClientSecurity(ClientDetails client) {
        // Validate redirect URIs
        for (String redirectUri : client.getRegisteredRedirectUri()) {
            if (!isSecureRedirectUri(redirectUri)) {
                throw new InvalidClientException("Insecure redirect URI: " + redirectUri);
            }
        }

        // Validate grant types
        if (client.getAuthorizedGrantTypes().contains("password")) {
            // Password grant should only be used by trusted first-party clients
            if (!client.getAuthorities().stream()
                    .anyMatch(auth -> auth.getAuthority().equals("ROLE_TRUSTED_CLIENT"))) {
                throw new InvalidClientException("Password grant not allowed for this client");
            }
        }
    }

    private boolean isSecureRedirectUri(String redirectUri) {
        // Must use HTTPS except for localhost development
        return redirectUri.startsWith("https://") ||
               redirectUri.startsWith("http://localhost") ||
               redirectUri.startsWith("http://127.0.0.1");
    }
}
```

**PKCE (Proof Key for Code Exchange) имплементация:**
```java
// PKCE Support за Enhanced Security
@Service
public class PKCEService {

    private final SecureRandom secureRandom = new SecureRandom();

    public PKCEChallenge generatePKCEChallenge() {
        // Generate code verifier (43-128 characters)
        String codeVerifier = generateCodeVerifier();

        // Generate code challenge using SHA256
        String codeChallenge = generateCodeChallenge(codeVerifier);

        return new PKCEChallenge(codeVerifier, codeChallenge, "S256");
    }

    public boolean validatePKCEChallenge(String codeVerifier, String codeChallenge,
                                        String challengeMethod) {
        if (!"S256".equals(challengeMethod)) {
            throw new InvalidGrantException("Unsupported PKCE challenge method");
        }

        String expectedChallenge = generateCodeChallenge(codeVerifier);
        return MessageDigest.isEqual(
            expectedChallenge.getBytes(StandardCharsets.UTF_8),
            codeChallenge.getBytes(StandardCharsets.UTF_8)
        );
    }

    private String generateCodeVerifier() {
        byte[] bytes = new byte[32]; // 256 bits
        secureRandom.nextBytes(bytes);
        return Base64.getUrlEncoder().withoutPadding().encodeToString(bytes);
    }

    private String generateCodeChallenge(String codeVerifier) {
        try {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            byte[] hash = digest.digest(codeVerifier.getBytes(StandardCharsets.UTF_8));
            return Base64.getUrlEncoder().withoutPadding().encodeToString(hash);
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException("SHA-256 not available", e);
        }
    }
}

// Enhanced Authorization Code Grant with PKCE
@RestController
@RequestMapping("/oauth")
public class PKCEAuthorizationController {

    @Autowired
    private PKCEService pkceService;

    @Autowired
    private AuthorizationCodeService authorizationCodeService;

    @PostMapping("/authorize")
    public ResponseEntity<Map<String, String>> authorize(
            @RequestParam String client_id,
            @RequestParam String redirect_uri,
            @RequestParam String scope,
            @RequestParam String state,
            @RequestParam String code_challenge,
            @RequestParam String code_challenge_method,
            Authentication authentication) {

        // Validate PKCE parameters
        if (!"S256".equals(code_challenge_method)) {
            throw new InvalidRequestException("code_challenge_method must be S256");
        }

        if (code_challenge == null || code_challenge.length() < 43) {
            throw new InvalidRequestException("Invalid code_challenge");
        }

        // Validate client и redirect URI
        ClientDetails client = clientDetailsService.loadClientByClientId(client_id);
        if (!client.getRegisteredRedirectUri().contains(redirect_uri)) {
            throw new InvalidRequestException("Invalid redirect_uri");
        }

        // Generate authorization code
        String authorizationCode = authorizationCodeService.createAuthorizationCode(
            client_id, authentication.getName(), scope, redirect_uri,
            code_challenge, code_challenge_method);

        // Build redirect URL
        String redirectUrl = String.format("%s?code=%s&state=%s",
                                          redirect_uri, authorizationCode, state);

        return ResponseEntity.ok(Map.of("redirect_url", redirectUrl));
    }
}
```

```java
    @Value("${jwt.refresh-token.expiry:604800}") // 7 days
    private int refreshTokenExpiry;

    public SecureJWTService(@Value("${jwt.secret}") String secretKey) {
        // Use HMAC-SHA512 for symmetric key
        this.signingKey = new SecretKeySpec(
            secretKey.getBytes(StandardCharsets.UTF_8),
            SignatureAlgorithm.HS512.getJcaName()
        );
        this.verificationKey = signingKey;
    }

    public TokenPair generateTokens(UserPrincipal userPrincipal) {
        Instant now = Instant.now();

        // Create access token with minimal claims
        String accessToken = Jwts.builder()
            .setIssuer(ISSUER)
            .setSubject(userPrincipal.getUsername())
            .setAudience("api")
            .setIssuedAt(Date.from(now))
            .setExpiration(Date.from(now.plusSeconds(accessTokenExpiry)))
            .setNotBefore(Date.from(now))
            .setId(UUID.randomUUID().toString()) // Unique token ID
            .claim("scope", userPrincipal.getAuthorities().stream()
                .map(GrantedAuthority::getAuthority)
                .collect(Collectors.toList()))
            .claim("type", "access")
            .claim("uid", userPrincipal.getUserId()) // User ID for faster lookups
            .signWith(signingKey, SignatureAlgorithm.HS512)
            .compact();

        // Create refresh token with different expiry
        String refreshToken = Jwts.builder()
            .setIssuer(ISSUER)
            .setSubject(userPrincipal.getUsername())
            .setAudience("refresh")
            .setIssuedAt(Date.from(now))
            .setExpiration(Date.from(now.plusSeconds(refreshTokenExpiry)))
            .setId(UUID.randomUUID().toString())
            .claim("type", "refresh")
            .claim("uid", userPrincipal.getUserId())
            .signWith(signingKey, SignatureAlgorithm.HS512)
            .compact();

        // Store refresh token за revocation capability
        storeRefreshToken(refreshToken, userPrincipal.getUsername());

        return new TokenPair(accessToken, refreshToken);
    }

    public Claims validateToken(String token) {
        try {
            Claims claims = Jwts.parserBuilder()
                .setSigningKey(verificationKey)
                .requireIssuer(ISSUER)
                .build()
                .parseClaimsJws(token)
                .getBody();

            // Additional security validations
            validateTokenSecurity(claims, token);

            return claims;

        } catch (JwtException e) {
            throw new InvalidTokenException("Invalid JWT token", e);
        }
    }

    private void validateTokenSecurity(Claims claims, String token) {
        // Check token blacklist
        if (isTokenBlacklisted(claims.getId())) {
            throw new InvalidTokenException("Token has been revoked");
        }

        // Validate token type
        String tokenType = claims.get("type", String.class);
        if (!"access".equals(tokenType) && !"refresh".equals(tokenType)) {
            throw new InvalidTokenException("Invalid token type");
        }

        // Check for token reuse (optional)
        if (detectTokenReuse(token, claims)) {
            // Log security incident
            securityEventService.logTokenReuseAttempt(claims.getSubject(), token);
            throw new InvalidTokenException("Potential token reuse detected");
        }

        // Validate token binding (optional)
        if (!validateTokenBinding(token, claims)) {
            throw new InvalidTokenException("Token binding validation failed");
        }
    }

    public TokenPair refreshTokens(String refreshToken) {
        Claims claims = validateToken(refreshToken);

        // Verify this is a refresh token
        if (!"refresh".equals(claims.get("type"))) {
            throw new InvalidTokenException("Invalid refresh token");
        }

        // Check if refresh token is still valid в database
        if (!isRefreshTokenValid(claims.getId(), claims.getSubject())) {
            throw new InvalidTokenException("Refresh token is no longer valid");
        }

        // Revoke old refresh token (refresh token rotation)
        revokeRefreshToken(claims.getId());

        // Load fresh user details
        UserPrincipal userPrincipal = userService.loadUserByUsername(claims.getSubject());

        // Generate new token pair
        return generateTokens(userPrincipal);
    }

    // Token revocation for logout
    public void revokeTokens(String refreshToken) {
        try {
            Claims claims = validateToken(refreshToken);
            String tokenId = claims.getId();

            // Add refresh token to blacklist
            revokeRefreshToken(tokenId);

            // Optionally blacklist associated access tokens
            blacklistUserTokens(claims.getSubject());

        } catch (Exception e) {
            // Log but don't fail - token might already be invalid
            logger.warn("Failed to revoke token", e);
        }
    }
}
```

```java
        String serviceToken = generateServiceToken();

        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.setBearerAuth(serviceToken);
        headers.set("X-Service-Name", serviceName);
        headers.set("X-Request-ID", generateRequestId());

        HttpEntity<Object> entity = new HttpEntity<>(request, headers);

        try {
            return restTemplate.exchange(
                serviceUrl + endpoint,
                HttpMethod.POST,
                entity,
                responseType
            );
        } catch (Exception e) {
            logger.error("Service call failed: {}", serviceUrl + endpoint, e);
            throw new ServiceCommunicationException("Failed to call service", e);
        }
    }

    private String generateServiceToken() {
        Instant now = Instant.now();

        return Jwts.builder()
            .setIssuer(serviceName)
            .setSubject("service")
            .setAudience("internal-services")
            .setIssuedAt(Date.from(now))
            .setExpiration(Date.from(now.plusSeconds(300))) // 5 minutes
            .claim("service_name", serviceName)
            .claim("type", "service_token")
            .signWith(Keys.hmacShaKeyFor(serviceSecret.getBytes()), SignatureAlgorithm.HS512)
            .compact();
    }

            .setDefaultRequestConfig(requestConfig)
            .setMaxConnTotal(100)
            .setMaxConnPerRoute(20)
            .build();

            new HttpComponentsClientHttpRequestFactory(httpClient);

        RestTemplate restTemplate = new RestTemplate(factory);

        // Add error handling
        restTemplate.setErrorHandler(new ServiceErrorHandler());

        return restTemplate;
```

```java
// API Gateway Rate Limiting
@Component
public class RateLimitingGatewayFilter implements GlobalFilter, Ordered {

    private final ReactiveRedisTemplate<String, String> redisTemplate;
    private final RateLimitingProperties rateLimitingProperties;

    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {
        ServerHttpRequest request = exchange.getRequest();
        String clientId = extractClientId(request);
        String endpoint = request.getPath().value();

        return checkRateLimit(clientId, endpoint)
            .flatMap(allowed -> {
                if (allowed) {
                    return chain.filter(exchange);
                } else {
                    return handleRateLimitExceeded(exchange);
                }
            });
    }

    private Mono<Boolean> checkRateLimit(String clientId, String endpoint) {
        String key = String.format("rate_limit:%s:%s", clientId, endpoint);
        long window = System.currentTimeMillis() / 60000; // 1-minute windows
        String windowKey = key + ":" + window;

        return redisTemplate.opsForValue().increment(windowKey)
            .flatMap(count -> {
                if (count == 1) {
                    // Set expiration for new window
                    return redisTemplate.expire(windowKey, Duration.ofMinutes(1))
                        .then(Mono.just(true));
                }

                RateLimitRule rule = rateLimitingProperties.getRuleForEndpoint(endpoint);
                return Mono.just(count <= rule.getRequestsPerMinute());
            });
    }

    private Mono<Void> handleRateLimitExceeded(ServerWebExchange exchange) {
        ServerHttpResponse response = exchange.getResponse();
        response.setStatusCode(HttpStatus.TOO_MANY_REQUESTS);
        response.getHeaders().add("Retry-After", "60");
        response.getHeaders().add("X-RateLimit-Limit", "100");
        response.getHeaders().add("X-RateLimit-Remaining", "0");

        String errorMessage = "{\"error\":\"Rate limit exceeded\",\"message\":\"Too many requests\"}";
        DataBuffer buffer = response.bufferFactory().wrap(errorMessage.getBytes());

        return response.writeWith(Mono.just(buffer));
    }

    @Override
    public int getOrder() {
        return -1; // Execute before other filters
    }
}
```
