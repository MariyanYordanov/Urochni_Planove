#### **Част 1: Системен мониторинг (20 минути)**

**Основни инструменти:**
```bash
# top - класически мониторинг
top

# htop - подобрен интерфейс (препоръчителен)
sudo apt install htop -y
htop

# Кратко обяснение на htop:
# - CPU bars показват натоварване
# - Mem/Swp показват памет
# - Tasks показват процеси
# - F9 = kill, F5 = tree view, F6 = sort
```

**Практика 1.1: Мониторинг на ресурси**
```bash
# CPU информация
lscpu
cat /proc/cpuinfo | grep "model name" | head -1

# Памет
free -h
cat /proc/meminfo | head -10

# Дисково пространство
df -h
du -sh /var/*

# Процеси по ресурси
ps aux --sort=-%mem | head -10    # Top 10 по памет
ps aux --sort=-%cpu | head -10    # Top 10 по CPU

# vmstat - виртуална памет статистика
vmstat 1 5    # На всяка 1 сек, 5 пъти

# iostat - I/O статистика
sudo apt install sysstat -y

```

**Практика 1.2: Мрежов мониторинг**

**Практика 1.3: Анализ на логове**

#### **Част 2: Backup стратегии (15 минути)**

**Теория - видове backup:**
```
┌─────────────────────────────────────────────────────────────┐
│                   ВИДОВЕ BACKUP                             │
├─────────────────────────────────────────────────────────────┤
│ FULL (Пълен)                                                │
│ • Копира ВСИЧКИ файлове                                     │
│ • Най-бавен и най-голям                                     │
│ • Най-лесен за възстановяване                               │
│ • Препоръка: седмично                                       │
├─────────────────────────────────────────────────────────────┤
│ INCREMENTAL (Инкрементален)                                 │
│ • Копира само ПРОМЕНЕНИТЕ от последния backup               │
│ • Най-бърз и най-малък                                      │
│ • За възстановяване: нужни всички инкременти               │
│ • Препоръка: дневно                                         │
├─────────────────────────────────────────────────────────────┤
│ DIFFERENTIAL (Диференциален)                                │
│ • Копира променените от последния FULL backup               │
│ • Средно бърз и среден размер                               │
│ • За възстановяване: full + последен differential          │
│ • Препоръка: дневно                                         │
└─────────────────────────────────────────────────────────────┘

Примерна седмична схема:
Нед: Full backup
Пон-Съб: Incremental (или Differential)
```

**3-2-1 правило:**
```
3-2-1 BACKUP ПРАВИЛО:
├── 3 копия на данните
│   └── Оригинал + 2 backup копия
├── 2 различни типа медии
│   └── Например: HDD + Cloud
└── 1 копие offsite
    └── Физически отделено (друга локация, cloud)

Защо offsite?
├── Защита от пожар
├── Защита от кражба
├── Защита от природни бедствия
└── Ransomware не може да достигне
```

#### **Част 3: Практически backup с rsync и tar (20 минути)**

**Практика 3.1: Tar backup**
```bash
# Създаване на тестови данни
mkdir -p ~/test_data
echo "Важни данни" > ~/test_data/important.txt
echo "Конфигурация" > ~/test_data/config.txt
mkdir ~/test_data/logs
echo "Log 1" > ~/test_data/logs/app.log

# Пълен backup с tar
tar -czvf backup_full.tar.gz ~/test_data/
# c = create
# z = gzip компресия
# v = verbose
# f = file

# Преглед на съдържанието
tar -tzvf backup_full.tar.gz

# Възстановяване
mkdir ~/restored
tar -xzvf backup_full.tar.gz -C ~/restored/

# С дата в името
DATE=$(date +%Y%m%d)
tar -czvf backup_${DATE}.tar.gz ~/test_data/
```

**Практика 3.2: Rsync основи**
```bash
# Rsync синтаксис:
# rsync [опции] източник дестинация

# Основни опции:
# -a  архивен режим (запазва права, времена, symlinks)
# -v  verbose
# -z  компресия
# -h  human-readable
# --delete  изтрива файлове в destination, които ги няма в source
# --dry-run  симулация без реално копиране

# Локална синхронизация
rsync -avh ~/test_data/ ~/backup_data/

# С --delete (внимавайте!)
rsync -avh --delete ~/test_data/ ~/backup_data/

# Симулация преди реално изпълнение
rsync -avh --delete --dry-run ~/test_data/ ~/backup_data/
```

**Практика 3.3: Rsync към отдалечен сървър**
```bash
# Към отдалечен сървър (чрез SSH)
rsync -avz ~/test_data/ user@server:/backup/

# От отдалечен сървър
rsync -avz user@server:/data/ ~/local_backup/

# С конкретен SSH порт
rsync -avz -e "ssh -p 2222" ~/test_data/ user@server:/backup/

# Exclude файлове
rsync -avz --exclude='*.log' --exclude='cache/' ~/test_data/ ~/backup/

# С bandwidth ограничение (KB/s)
rsync -avz --bwlimit=1000 ~/test_data/ user@server:/backup/
```

**Практика 3.4: Автоматизиран backup скрипт**
```bash
#!/bin/bash
# backup_system.sh - Автоматизиран backup скрипт

# Конфигурация
SOURCE="/home/$USER"
BACKUP_DIR="/backup"
REMOTE_HOST="backup-server"
REMOTE_PATH="/backups/$HOSTNAME"
LOG_FILE="/var/log/backup.log"
DATE=$(date +%Y%m%d_%H%M%S)
DAY_OF_WEEK=$(date +%u)  # 1-7, 1=понеделник

# Функции
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" >> "$LOG_FILE"
    echo "$1"
}

# Създаване на backup директория
mkdir -p "$BACKUP_DIR"

log "===== BACKUP STARTED ====="

# Определяне на типа backup
if [ "$DAY_OF_WEEK" -eq 7 ]; then
    # Неделя - Full backup
    BACKUP_TYPE="full"
    BACKUP_FILE="backup_full_$DATE.tar.gz"
    log "Тип: FULL backup"
    
    tar -czf "$BACKUP_DIR/$BACKUP_FILE" \
        --exclude='*.tmp' \
        --exclude='cache' \
        --exclude='.cache' \
        "$SOURCE" 2>> "$LOG_FILE"
else
    # Делничен ден - Incremental
    BACKUP_TYPE="incremental"
    BACKUP_FILE="backup_incr_$DATE.tar.gz"
    SNAPSHOT_FILE="$BACKUP_DIR/snapshot.snar"
    log "Тип: INCREMENTAL backup"
    
    tar -czf "$BACKUP_DIR/$BACKUP_FILE" \
        --listed-incremental="$SNAPSHOT_FILE" \
        --exclude='*.tmp' \
        "$SOURCE" 2>> "$LOG_FILE"
fi

# Проверка на успеха
if [ $? -eq 0 ]; then
    SIZE=$(ls -lh "$BACKUP_DIR/$BACKUP_FILE" | awk '{print $5}')
    log "Backup създаден успешно: $BACKUP_FILE ($SIZE)"
    
    # Синхронизация към отдалечен сървър (ако е достъпен)
    if ping -c 1 "$REMOTE_HOST" &> /dev/null; then
        log "Синхронизация към $REMOTE_HOST..."
        rsync -az "$BACKUP_DIR/$BACKUP_FILE" "$REMOTE_HOST:$REMOTE_PATH/"
        if [ $? -eq 0 ]; then
            log "Синхронизация успешна"
        else
            log "ГРЕШКА: Синхронизацията неуспешна"
        fi
    else
        log "ВНИМАНИЕ: $REMOTE_HOST недостъпен, пропускане на синхронизация"
    fi
else
    log "ГРЕШКА: Backup неуспешен!"
    exit 1
fi

# Почистване на стари backup-и (по-стари от 30 дни)
log "Почистване на стари backup-и..."
find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +30 -delete
log "Почистването завършено"

log "===== BACKUP COMPLETED ====="
```
