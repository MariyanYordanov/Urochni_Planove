#### **Част 2: Условия и цикли (15 минути)**

**Условни конструкции:**
```bash
#!/bin/bash
# conditions.sh

# If-else
read -p "Въведете число: " NUM

if [ $NUM -gt 10 ]; then
    echo "$NUM е по-голямо от 10"
elif [ $NUM -eq 10 ]; then
    echo "$NUM е равно на 10"
else
    echo "$NUM е по-малко от 10"
fi

# Сравнения за числа:
# -eq  равно
# -ne  различно
# -gt  по-голямо
# -lt  по-малко
# -ge  по-голямо или равно
# -le  по-малко или равно

# Сравнения за низове:
# =    равни
# !=   различни
# -z   празен низ
# -n   непразен низ

# Проверки за файлове:
# -f   съществува и е файл
# -d   съществува и е директория
# -e   съществува
# -r   четим
# -w   записваем
# -x   изпълним
```

**Практика 2.1: Проверка на файл**
```bash
#!/bin/bash
# file_check.sh

FILE=$1

if [ -z "$FILE" ]; then
    echo "Употреба: $0 <файл>"
    exit 1
fi

if [ -f "$FILE" ]; then
    echo "$FILE е файл"
    echo "Размер: $(ls -lh $FILE | awk '{print $5}')"
elif [ -d "$FILE" ]; then
    echo "$FILE е директория"
    echo "Съдържа: $(ls $FILE | wc -l) елемента"
else
    echo "$FILE не съществува"
    exit 1
fi
```

**Цикли:**
```bash
#!/bin/bash
# loops.sh

# For цикъл - списък
echo "=== For с списък ==="
for FRUIT in apple banana cherry; do
    echo "Плод: $FRUIT"
done

# For цикъл - файлове
echo "=== For с файлове ==="
for FILE in /etc/*.conf; do
    echo "Конфиг: $FILE"
done

# For цикъл - числа
echo "=== For с числа ==="
for i in {1..5}; do
    echo "Число: $i"
done

# While цикъл
echo "=== While ==="
COUNT=1
while [ $COUNT -le 5 ]; do
    echo "Count: $COUNT"
    ((COUNT++))
done

# Until цикъл
echo "=== Until ==="
NUM=1
until [ $NUM -gt 5 ]; do
    echo "Num: $NUM"
    ((NUM++))
done
```

**Практика 2.2: Обработка на файлове**
```bash
#!/bin/bash
# process_files.sh

DIRECTORY=${1:-.}  # Default: текуща директория

echo "Обработка на файлове в $DIRECTORY"
echo "================================="

for FILE in "$DIRECTORY"/*; do
    if [ -f "$FILE" ]; then
        FILENAME=$(basename "$FILE")
        SIZE=$(stat -c%s "$FILE")
        echo "$FILENAME - $SIZE bytes"
    fi
done

echo "================================="
echo "Общо: $(ls -1 "$DIRECTORY" | wc -l) елемента"
```

**Практика 3.1: Backup скрипт**
```bash
#!/bin/bash
# backup.sh - Скрипт за backup

# Конфигурация
SOURCE_DIR="/home/$USER/Documents"
BACKUP_DIR="/home/$USER/Backups"
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_NAME="backup_$DATE.tar.gz"

# Функция за логване
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1"
}

# Създаване на backup директория
mkdir -p "$BACKUP_DIR"

# Проверка дали source съществува
if [ ! -d "$SOURCE_DIR" ]; then
    log "ГРЕШКА: $SOURCE_DIR не съществува!"
    exit 1
fi

# Създаване на архив
log "Започване на backup..."
tar -czf "$BACKUP_DIR/$BACKUP_NAME" -C "$(dirname $SOURCE_DIR)" "$(basename $SOURCE_DIR)"

if [ $? -eq 0 ]; then
    log "Backup създаден: $BACKUP_DIR/$BACKUP_NAME"
    log "Размер: $(ls -lh $BACKUP_DIR/$BACKUP_NAME | awk '{print $5}')"
else
    log "ГРЕШКА при създаване на backup!"
    exit 1
fi

# Изтриване на стари backup-и (по-стари от 7 дни)
log "Почистване на стари backup-и..."
find "$BACKUP_DIR" -name "backup_*.tar.gz" -mtime +7 -delete

log "Готово!"
```

#### **Част 4: Cron автоматизация (15 минути)**

**Cron синтаксис:**
```
┌───────────── минута (0 - 59)
│ ┌───────────── час (0 - 23)
│ │ ┌───────────── ден от месеца (1 - 31)
│ │ │ ┌───────────── месец (1 - 12)
│ │ │ │ ┌───────────── ден от седмицата (0 - 7, 0 и 7 = неделя)
│ │ │ │ │
* * * * * команда

Примери:
* * * * *         # Всяка минута
0 * * * *         # Всеки час (на :00)
0 0 * * *         # Всеки ден в полунощ
0 2 * * *         # Всеки ден в 02:00
30 4 1 * *        # В 04:30 на 1-во число
0 0 * * 0         # Всяка неделя в полунощ
*/15 * * * *      # На всеки 15 минути
0 9-17 * * 1-5    # Всеки час 9-17, понеделник-петък
```

**Практика 4.1: Управление на crontab**
```bash
# Редактиране на crontab
crontab -e

# Преглед на текущите задачи
crontab -l

# Изтриване на всички задачи
crontab -r

# Редактиране на crontab на друг потребител (root)
sudo crontab -u username -e

# Системни cron директории
ls /etc/cron.d/
ls /etc/cron.daily/
ls /etc/cron.hourly/
ls /etc/cron.weekly/
ls /etc/cron.monthly/
```

**Практика 4.2: Добавяне на cron задачи**
```bash
# Редактирайте crontab
crontab -e

# Добавете редове:
# ─────────────────────────────────────────
# Backup всеки ден в 02:00
0 2 * * * /home/student/scripts/backup.sh >> /var/log/backup.log 2>&1

# Почистване на temp файлове всяка неделя
0 3 * * 0 find /tmp -type f -mtime +7 -delete

# Проверка на дисково пространство на всеки час
0 * * * * df -h > /home/student/disk_status.txt

# Изпращане на системен статус всеки ден в 08:00
0 8 * * * /home/student/scripts/system_status.sh | mail -s "Daily Report" admin@example.com
# ─────────────────────────────────────────
```

### **4. ЗАТВЪРЖДАВАНЕ (12 минути)**

**Практическа задача:**
```
Създайте скрипт "monitor.sh", който:

1. Проверява дали услуга (nginx, ssh) работи
2. Ако не работи - опитва да я рестартира
3. Записва резултата в лог файл
4. Изпраща предупреждение (echo) ако рестартът не успее

5. Добавете скрипта в cron да се изпълнява на всеки 5 минути
```

**Примерно решение:**
```bash
#!/bin/bash
# monitor.sh

SERVICE=${1:-nginx}
LOG_FILE="/var/log/monitor_$SERVICE.log"

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" >> "$LOG_FILE"
}

if systemctl is-active --quiet "$SERVICE"; then
    log "$SERVICE работи нормално"
else
    log "ВНИМАНИЕ: $SERVICE не работи!"
    log "Опит за рестартиране..."
    
    sudo systemctl restart "$SERVICE"
    
    sleep 2
    
    if systemctl is-active --quiet "$SERVICE"; then
        log "$SERVICE рестартиран успешно"
    else
        log "КРИТИЧНО: $SERVICE не може да бъде рестартиран!"
        echo "ALERT: $SERVICE is down and could not be restarted!"
    fi
fi

# Crontab:
# */5 * * * * /home/student/scripts/monitor.sh nginx
```

### ПРИЛОЖЕНИЕ 1: BASH СИНТАКСИС СПРАВОЧНИК

```
╔══════════════════════════════════════════════════════════════╗
║                    BASH СПРАВОЧНИК                           ║
╚══════════════════════════════════════════════════════════════╝

ПРОМЕНЛИВИ
───────────────────────────────────────────────────────────────
VAR="value"                 # Присвояване
$VAR или ${VAR}             # Използване
${VAR:-default}             # Default стойност
${#VAR}                     # Дължина на низ
$0, $1, $2...               # Аргументи
$#                          # Брой аргументи
$@                          # Всички аргументи
$?                          # Exit code
$$                          # PID на скрипта

УСЛОВИЯ
───────────────────────────────────────────────────────────────
if [ condition ]; then      # If
elif [ condition ]; then    # Else if
else                        # Else
fi                          # Край

# Числа: -eq -ne -gt -lt -ge -le
# Низове: = != -z -n
# Файлове: -f -d -e -r -w -x

ЦИКЛИ
───────────────────────────────────────────────────────────────
for VAR in list; do         # For
    commands
done

while [ condition ]; do     # While
    commands
done

ФУНКЦИИ
───────────────────────────────────────────────────────────────
function_name() {
    local VAR=$1
    commands
    return value
}

ПОЛЕЗНИ КОНСТРУКЦИИ
───────────────────────────────────────────────────────────────
$(command)                  # Command substitution
$((expression))             # Аритметика
[ ] или [[ ]]               # Test условия
&& ||                       # Логически оператори
| > >> <                    # Пренасочване

═══════════════════════════════════════════════════════════════
```
